[docs]def make_conn_str(file_path: Path) -> str:
    """
    Create a SQLite Conn String

    Parameters
    ----------
    file_path: Path
        Fiole path of the SQLite file

    Returns
    -------
    str
        SQLAlchemy connection string
    """
    return str(
        sqlalchemy.engine.url.URL.create(drivername="sqlite", host="/" + str(file_path))
    )
